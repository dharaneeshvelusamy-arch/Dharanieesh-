#include <stdio.h>

struct student {
    int roll;
    char name[50];
    int m1, m2, m3;
    int total;
    float average;
};

int main() {
    struct student s[100];
    int n, i;

    printf("Enter number of students: ");
    scanf("%d", &n);

    for (i = 0; i < n; i++) {
        printf("\nEnter details of student %d\n", i + 1);

        printf("Roll Number: ");
        scanf("%d", &s[i].roll);

        printf("Name: ");
        scanf(" %[^\n]", s[i].name);

        printf("Enter 3 subject marks: ");
        scanf("%d %d %d", &s[i].m1, &s[i].m2, &s[i].m3);

        s[i].total = s[i].m1 + s[i].m2 + s[i].m3;
        s[i].average = s[i].total / 3.0;
    }

    printf("\n\n===== STUDENT ACADEMIC DETAILS =====\n");
    printf("Roll\tName\t\tTotal\tAverage\tResult\n");

    for (i = 0; i < n; i++) {
        printf("%d\t%s\t\t%d\t%.2f\t%s\n",
               s[i].roll,
               s[i].name,
               s[i].total,
               s[i].average,
               (s[i].average >= 40) ? "PASS" : "FAIL");
    }

    return 0;
}
